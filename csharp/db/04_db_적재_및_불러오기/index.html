using System.Windows.Forms;

namespace UNIT_TEST_DB
{
    public partial class DB_TESTER : Form
    {
        private List<string[]> masterList;
        private MariaDB mariaDB;
        public OpenAPI_DB()
        {
            InitializeComponent();
            mariaDB = new MariaDB(~~ìƒëµ~~);
        }
        
        private void btnInsertAuto_Click(object sender, EventArgs e)
        {
            mariaDB.connectDBMS();
            Console.WriteLine("DBMS State : " + mariaDB.DBMSstate);

            string queryPreString = "INSERT INTO [tableName] VALUES";
            string queryPostString = "";
            string now = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss");
            foreach (string[] list in masterList)
            {
                queryPostString += string.Format("('{0}','{1}','{2}'),", list[0], list[1], now);
            }
            queryPostString = queryPostString[..^1];

            string queryString = queryPreString + " " + queryPostString + ";";
            Console.WriteLine(queryString);

            mariaDB.setData(queryString);

            mariaDB.disconnectDBMS();
            Console.WriteLine("DBMS State : " + mariaDB.DBMSstate);
        }

        private void btnSql_Click(object sender, EventArgs e)
        {
            mariaDB.connectDBMS();
            Console.WriteLine("DBMS State : " + mariaDB.DBMSstate);

            DataSet ds = mariaDB.getData_dataset(this.txtSql.Text);	// SELECT * FROM [tableName]
            this.dataGridView1.DataSource = ds.Tables[0];

            mariaDB.disconnectDBMS();
            Console.WriteLine("DBMS State : " + mariaDB.DBMSstate);
        }
    }
}
