    using System;
using System.Collections.Generic;
using System.Data;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using MySql.Data.MySqlClient;

namespace UNIT_TEST_DB
{
    class MariaDB
    {
        private MySqlConnection conn = null;
        private MySqlCommand command = null;
        private string connString;
        private string server;
        private int port;
        private string db;
        private string id;
        private string pwd;

        public MariaDB(string server, int port, string db, string id, string pwd)
        {
            this.server = server;
            this.port = port;
            this.db = db;
            this.id = id;
            this.pwd = pwd;

            this.connString = string.Format("Server={0};Port={1};Database={2};Uid={3};Pwd={4};",
                                           this.server, this.port.ToString(), this.db, this.id, this.pwd);
            this.conn = new MySqlConnection(connString);
        }
        ~MariaDB()
        {
            conn.Close();
            conn.Dispose();
        }
        public void connectDBMS()
        {
            try
            {
                if (conn == null)
                {
                    conn = new MySqlConnection(connString);
                }
                if (conn.State == ConnectionState.Closed)
                {
                    conn.Open();
                }
            }
            catch (Exception connErr)
            {
                Console.WriteLine("connErr : " + connErr);
            }
        }
        public void disconnectDBMS()
        {
            if (DBMSstate == ConnectionState.Open)
            {
                conn.Close();
            }
        }

        public Array connInfo
        {
            get { return new String[] { this.server, this.port.ToString(), this.db, this.id, this.pwd }; }
        }
        public ConnectionState DBMSstate
        {
            get { return conn.State; }
        }

        public void setData(string queryString)
        {
            conn.Open();
            command = new MySqlCommand(queryString, conn);
            try
            {
                command.ExecuteNonQuery();
            }
            catch (MySqlException sqlSetErr)
            {
                Console.WriteLine("sqlSetErr : " + sqlSetErr);
            }
            conn.Close();
        }

        public DataSet getData_dataset(string queryString)
        {
            conn.Open();
            DataSet ds = new DataSet("mainSession");
            command = new MySqlCommand(queryString, conn);
            using (MySqlDataAdapter adapter = new MySqlDataAdapter(command))
            {
                adapter.Fill(ds, "userTable");
            }
            conn.Close();
            return ds;
        }
        public List<string> getData_list(string queryString)
        {
            conn.Open();
            List<string> fieldNames = new List<string>();
            List<string> datas = new List<string>();
            try
            {
                command = new MySqlCommand(queryString, conn);
                using (MySqlDataReader reader = command.ExecuteReader())
                {
                    for (int fieldCnt = 0; fieldCnt < reader.FieldCount; fieldCnt++)
                    {
                        fieldNames.Add(reader.GetName(fieldCnt));
                    }
                    while (reader.Read())
                    {
                        for (int fieldCnt = 0; fieldCnt < reader.FieldCount; fieldCnt++)
                        {
                            datas.Add(reader[fieldCnt].ToString());
                        }
                    }
                }
            }
            catch (MySqlException sqlGetErr)
            {
                Console.WriteLine("sqlGetErr : " + sqlGetErr);
            }
            conn.Close();
            return datas;
        }
    }
}
