    using System;
using System.Windows.Forms;
using MySql.Data.MySqlClient;					// Load;

namespace UNIT_TEST_DB
{
    public partial class MainForm : Form
    {
        public MainForm()
        {
            InitializeComponent();
        }
        ~MainForm()
        {
            conn.Close();
            conn.Dispose();
        }

        MySqlConnection conn;			// local variable;

        private bool connectMariaDB()
        {
            string connStr = string.Format("Server={0};Port={1};Database={2};Uid={3};Pwd={4};",
this.txtServer.Text, "3306", this.txtDB.Text, this.txtId.Text, this.txtPwd.Text);
            try
            {
                conn = new MySqlConnection(connStr);	// connection;
                conn.Open();
                return true;
            }
            catch(Exception e)
            {
                Console.WriteLine(e);
                return false;
            }
        }
        private void btnClose_Click(object sender, EventArgs e)
        {
            this.Close();
        }

        private void btnConn_Click(object sender, EventArgs e)
        {
            if (connectMariaDB() == true)			// Print;
            {
                this.lblStatus.Text = "Success";
            }
            else
            {
                this.lblStatus.Text = "Fail";
            }
        }

        private void btnSql_Click(object sender, EventArgs e)
        {
            string sqlStr = this.txtSql.Text;
            try
            {
                MySqlCommand command = new MySqlCommand(sqlStr, conn);
                command.ExecuteNonQuery();			// send Sql String;
            }
            catch (Exception ex)
            {
                Console.WriteLine(ex); 
            }
        }
    }
}
