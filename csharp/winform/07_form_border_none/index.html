    namespace UNIT_TEST_GLOBAL
{
    public partial class Form1 : Form
    {
        TestAPI testAPI = new TestAPI(3406);
        public Form1()
        {
            InitializeComponent();
            StartSubForm();
        }

        private void StartSubForm()
        {
            SubForm sub = new SubForm();
            sub.testAPI = this.testAPI;
            TextBox txtBox = new TextBox()
            {
                Text = "test sample text",
                Location = new Point(10, 10),
                Size = new Size(20, 60),
            };
            sub.txtTest = txtBox;
            sub.Controls.Add(txtBox);

            sub.Load += (sender, e) =>
            {
                Console.WriteLine($"In SubForm Id: {sub.testAPI.Id}");
                Console.WriteLine((this.testAPI == sub.testAPI).ToString()) ;
                Console.WriteLine();

                sub.testAPI.Id = 1999;
                Console.WriteLine($"In Form1 Id: {this.testAPI.Id}");
                Console.WriteLine($"In SubForm Id: {sub.testAPI.Id}");
                Console.WriteLine((this.testAPI == sub.testAPI).ToString()) ;

                Console.WriteLine(sub.txtTest.Text);

            };
            sub.FormClosed += (sender, e) =>
            {
                Console.WriteLine();
                Console.WriteLine("subForm Closed.");
                Console.WriteLine($"In Form1 Id: {this.testAPI.Id}");
                Console.WriteLine($"In SubForm Id: {sub.testAPI.Id}");
                Console.WriteLine(sub.txtTest.Text);

            };
            sub.Show();
        }
    }
    public class TestAPI
    {
        public int Id { get; set; }
        public TestAPI(int Id)
        {
            this.Id = Id;
            Console.WriteLine($"Success To Init value. [{this.Id}]");
        }
    }
}
